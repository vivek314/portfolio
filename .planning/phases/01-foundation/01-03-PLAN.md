    ```typescript
    // src/lib/auth.ts
    // All server-side auth operations go through this file.
    // Import THIS file in Server Components — NOT '@clerk/nextjs/server' directly.
    // This ensures consistent import patterns and makes future auth changes a single-file edit.
    import { auth, currentUser } from '@clerk/nextjs/server'
    import { redirect } from 'next/navigation'

    /**
     * Get current userId or null if not authenticated.
     * Use for optional auth checks (e.g., showing different UI for logged-in vs. guest).
     */
    export async function getCurrentUserId(): Promise<string | null> {
      const { userId } = await auth()
      return userId
    }

    /**
     * Require authentication — redirect to /sign-in if not authenticated.
     * Use this in every protected Server Component and Route Handler.
     */
    export async function requireAuth(): Promise<string> {
      const { userId } = await auth()
      if (!userId) redirect('/sign-in')
      return userId
    }

    /**
     * Get full Clerk user object for the current session.
     * Only use when you need user profile data (name, email) — more expensive than auth().
     */
    export { currentUser }
    ```
