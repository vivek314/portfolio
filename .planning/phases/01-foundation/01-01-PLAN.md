    ```typescript
    // These are the core domain types — keep them stable; downstream phases depend on them.

    export type Difficulty = 'easy' | 'medium' | 'hard'

    export type ScenarioId = 'parking-lot'  // Extensible: add 'url-shortener' etc. in Phase 5+

    export interface ScenarioConfig {
      id: ScenarioId
      name: string
      difficulty: Difficulty
    }

    // Submission stored in Neon — matches Drizzle schema in Plan 02
    export interface Submission {
      id: string
      clerkUserId: string
      scenarioId: ScenarioId
      difficulty: Difficulty
      code: string
      createdAt: Date
    }

    // Per-criterion evaluation result — used in Plans 04 (evaluator) + 06 (dashboard)
    export interface CriterionResult {
      criterionId: string
      criterionName: string
      passed: boolean
      explanation: string
    }

    // Full evaluation result — references a submission
    export interface EvaluationResult {
      id: string
      submissionId: string
      criteriaResults: CriterionResult[]
      overallPass: boolean
      llmModel: string
      createdAt: Date
    }
    ```
